#Cluster function
#hclust  -  it requires a distance object be input

#Normalize/Standardize data

#install.packages("clusterSim")
library(clusterSim)
food.std <-
    round(data.Normalization(food[,-1], type = "n1", normalization = "column"),
          2)

# Arguments
# n0 - without normalization
# n1 - standardization ((x-mean)/sd)

#Distance measure - Euclidian distances
#Calculating pairwise Euclidean distances between the (standardized) objects:
dist.food <- round(dist(food.std), 2)
dist_food_matrix <- as.matrix(dist.food)

#Ward Method
food.ward.link <- hclust(d = dist.food, method = "ward.D2")
food.ward.link

# Plotting the ward linkage dendrogram:
plclust(tree = food.ward.link, ylab = "Distance")

plclust(tree = food.ward.link,
        labels = Food,
        ylab = "Distance")

plclust(
    tree = food.ward.link,
    labels = Food,
    ylab = "Distance",
    hang = -1
)

#'plclust' is deprecated use 'plot' instead
plot(x = food.ward.link, ylab = "Distance", xlab = "Food")

plot(
    x = food.ward.link,
    ylab = "Distance",
    xlab = "Food",
    hang = -1
)

plot(
    x = food.ward.link,
    ylab = "Distance",
    xlab = "Food",
    hang = -1,
    labels = Food
)

#The fixed classification can be visually demonstrated with
#rect.hclust function

plot(
    x = food.ward.link,
    ylab = "Distance",
    xlab = "Food",
    hang = -1,
    labels = Food
)

rect.hclust(tree = food.ward.link, k = 3)

#We can extract classification in a certain level using function cutree
food.ward.cutree.by3Clust <- cutree(food.ward.link, k = 3)
food.ward.cutree.by3Clust  # printing the "clustering vector"

food.ward.clust3 <-
    lapply(1:3, function(nc)
        Food[food.ward.cutree.by3Clust == nc])
food.ward.clust3  # printing the clusters in terms of the Food labels

table(food.ward.cutree.by3Clust)

food.ward.cutree.byHeight6 <- cutree(food.ward.link, h = 6)
food.ward.cutree.byHeight6  # printing the "clustering vector"

food.ward.clust.byHeight6 <-
    lapply(1:3, function(nc)
        Food[food.ward.cutree.byHeight6 == nc])
food.ward.clust.byHeight6   # printing the clusters in terms of the Food labels

table(food.ward.cutree.byHeight6)
